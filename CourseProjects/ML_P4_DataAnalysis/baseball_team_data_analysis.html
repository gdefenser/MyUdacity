def drawGrid(set1,set2,x_key,y_key,label_1,label_2):
    plt.plot(set1[x_key], set1[y_key], 'r', label=label_1)
    plt.plot(set2[x_key], set2[y_key], 'b', label=label_2)
    plt.legend(bbox_to_anchor=[0.4, 1])
    ax = plt.gca()
    ax.set_ylabel(y_key)
    ax.set_xlabel(x_key)
    plt.grid()
